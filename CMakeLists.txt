cmake_minimum_required(VERSION 3.6)

project(gfc VERSION 1.0.0 DESCRIPTION "g foundation class")

set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -D_BSD_SOURCE -DGFC_USE_OPENSSL -Wno-format -Wno-deprecated-declarations")

option(WITH_GFC_JVM "gfc_jvm.c" ON)
option(WITH_GFC_CRYPTO "gfc_crypto.c" ON)

set(GFC_SOURCES 
	src/gfc_base64.c
  src/gfc_hex.c
  src/gfc_list.c
  src/gfc_log.c
  src/gfc_lru.c
  src/gfc_map.c
  src/gfc_ring.c
  src/gfc_sha1.c
  src/gfc_string.c
  src/gfc_tcp.c
  src/gfc_utf8.c
  src/gfc_uuid.c
  src/gfc_date.c
  src/gfc_btree.c
  src/gfc_mt.c
  src/gfc_ws.c
  src/gfc_gc.c
  src/gfc_fs.c
  src/gfc_cli.c
)

####################
##
## WITH OPTIONS
##
####################
if(WITH_GFC_JVM)
	list(APPEND GFC_SOURCES src/gfc_jvm.c)
endif(WITH_GFC_JVM)
if(WITH_GFC_CRYPTO)
	list(APPEND GFC_SOURCES src/gfc_crypto.c)
endif(WITH_GFC_CRYPTO)

####################
##
## LINK
##
####################
if(UNIX)
  link_directories(
	  /opt/local/lib
	)
endif(UNIX)

####################
##
## INCLUDE
##
####################
if (WIN32)
	include_directories(
		"include"
	)
else ()  
	include_directories(
		"/usr/local/include"
		"/opt/local/include"
		"include"
	)
	if(WITH_GFC_JVM)	
		include_directories(
			"/Library/Java/JavaVirtualMachines/jdk1.8.0_212.jdk/Contents/Home/include"
			"/Library/Java/JavaVirtualMachines/jdk1.8.0_212.jdk/Contents/Home/include/darwin"
		)
	endif()
endif(WIN32)

add_library(gfc STATIC ${GFC_SOURCES})

if (WIN32)
	link_directories(
		"/opt/local/x86_64-w64-mingw32/lib"
	)
	target_link_libraries(gfc PRIVATE
	  libiconv.dll.a
	)
else()
	if(WITH_GFC_JVM)	
		link_directories(
		"/Library/Java/JavaVirtualMachines/jdk1.8.0_212.jdk/Contents/Home/jre/lib/server/"
		"/Library/Java/JavaVirtualMachines/jdk1.8.0_212.jdk/Contents/Home/jre/lib/jli/"
		)
		target_link_libraries(gfc PRIVATE
		  jli
		  jvm
		)	
	endif()
endif (WIN32)
